close =  np.array([float(x) for x in bitCoin_data['Value']])

bitCoin_data['SMA10'].fillna(method='bfill', inplace=True)
bitCoin_data['SMA20'].fillna(method='bfill', inplace=True)
bitCoin_data['SMA30'].fillna(method='bfill', inplace=True)

# 可视化

        return ''
    return bitCoin_data['Date'][int(x)]
ax.xaxis.set_major_formatter(ticker.FuncFormatter(format_date))

fig = plt.figure(figsize=(24, 16))
ax = fig.add_subplot(111)

dif, dea, bar = talib.MACD(close, fastperiod=12, slowperiod=26, signalperiod=9)
dif[np.isnan(dif)], dea[np.isnan(dea)], bar[np.isnan(bar)] = 0, 0, 0
bitCoin_data['MACD'], bitCoin_data['MACDsignal'], bitCoin_data['MACDhist'] = dif, dea, bar

tmp_len = 300 # 控制绘制的天数
ax.plot(np.arange(0, tmp_len), dif[:tmp_len])
ax.plot(np.arange(0, tmp_len), dea[:tmp_len])
red_bar = np.where(bar > 0, 2 * bar, 0)
blue_bar = np.where(bar < 0, 2 * bar, 0)
ax.bar(np.arange(0, tmp_len), red_bar[:tmp_len], color="red")
ax.bar(np.arange(0, tmp_len), blue_bar[:tmp_len], color="blue")

    # 由于前面股票数据在 date 这个位置传入的都是int

        return ''
